## 🎨 Design Spec (Sati)

### Fluxo de interação

```
/grafico
  └─ Resposta: embed "📊 Qual gráfico?" + GraficoTypeView
       ├─ [TVL]         → PeriodoView(chart_type="tvl")
       ├─ [P&L]         → PeriodoView(chart_type="pnl")
       └─ [Operações]   → PeriodoView(chart_type="ops")
            ├─ [7 dias]  → gera chart + embed
            ├─ [30 dias] → gera chart + embed
            └─ [90 dias] → gera chart + embed
```

### `chart_views.py` (estrutura)

```python
import discord
from discord.ui import View, Button
from .chart_handler import generate_chart
from .embed_builder import build_embed
from .design_tokens import PINK_LIGHT, CHART_BLUE

CHART_LABELS = {"tvl": "TVL", "pnl": "P&L", "ops": "Operações"}
PERIOD_OPTIONS = [("7 dias", 7), ("30 dias", 30), ("90 dias", 90)]

class PeriodoView(View):
    def __init__(self, chart_type: str):
        super().__init__(timeout=120)
        self.chart_type = chart_type
        for label, days in PERIOD_OPTIONS:
            btn = Button(label=label, style=discord.ButtonStyle.secondary)
            btn.callback = self._make_callback(days)
            self.add_item(btn)

    def _make_callback(self, days):
        async def callback(interaction):
            await interaction.response.defer()
            buf = generate_chart(self.chart_type, days)
            label = CHART_LABELS[self.chart_type]
            embed = build_embed(
                "chart",
                title=f"📊 {label} — Últimos {days} dias",
                description=f"Dados do protocolo WEbdEX",
            )
            await interaction.edit_original_response(
                embed=embed,
                attachments=[discord.File(buf, filename="chart.png")],
                view=None,
            )
        return callback

    async def on_timeout(self):
        for item in self.children:
            item.disabled = True

class GraficoTypeView(View):
    def __init__(self):
        super().__init__(timeout=120)
```
