### docker-compose.yml — serviços desta story

```yaml
version: '3.8'

networks:
  orchestrator_net:
    external: false

volumes:
  postgres_data:
  redis_data:
  n8n_data:

services:
  redis:
    image: redis:7-alpine
    restart: unless-stopped
    volumes:
      - redis_data:/data
    networks:
      - orchestrator_net
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  postgres:
    image: pgvector/pgvector:pg16
    restart: unless-stopped
    environment:
      POSTGRES_DB: orchestrator
      POSTGRES_USER: orchestrator
      POSTGRES_PASSWORD: ${POSTGRES_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./infra/postgres/init.sql:/docker-entrypoint-initdb.d/init.sql
    networks:
      - orchestrator_net
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U orchestrator"]
      interval: 10s
      timeout: 5s
      retries: 5

  n8n:
    image: n8nio/n8n:latest
    restart: unless-stopped
    environment:
      N8N_BASIC_AUTH_ACTIVE: "true"
      N8N_BASIC_AUTH_USER: ${N8N_USER}
      N8N_BASIC_AUTH_PASSWORD: ${N8N_PASSWORD}
      N8N_HOST: ${N8N_HOST}
      N8N_PORT: 5678
      N8N_PROTOCOL: https
      WEBHOOK_URL: https://${N8N_HOST}
      DB_TYPE: postgresdb
      DB_POSTGRESDB_HOST: postgres
      DB_POSTGRESDB_PORT: 5432
      DB_POSTGRESDB_DATABASE: orchestrator
      DB_POSTGRESDB_USER: orchestrator
      DB_POSTGRESDB_PASSWORD: ${POSTGRES_PASSWORD}
    volumes:
      - n8n_data:/home/node/.n8n
    networks:
      - orchestrator_net
    depends_on:
      postgres:
        condition: service_healthy
    labels:
      - "traefik.enable=true"
      - "traefik.http.routers.n8n.rule=Host(`${N8N_HOST}`)"
      - "traefik.http.routers.n8n.tls=true"
      - "traefik.http.routers.n8n.tls.certresolver=letsencrypt"
      - "traefik.http.services.n8n.loadbalancer.server.port=5678"
```

### infra/postgres/init.sql — schema inicial

```sql
-- Habilita pgvector
CREATE EXTENSION IF NOT EXISTS vector;

-- Tabela de eventos recebidos de qualquer plataforma
CREATE TABLE IF NOT EXISTS platform_events (
    id          BIGSERIAL PRIMARY KEY,
    platform    VARCHAR(20) NOT NULL,  -- discord | instagram | whatsapp | telegram
    event_type  VARCHAR(50) NOT NULL,  -- message | mention | reaction | dm
    payload     JSONB       NOT NULL,
    processed   BOOLEAN     DEFAULT FALSE,
    created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Tabela de mensagens processadas pelo AI Responder
CREATE TABLE IF NOT EXISTS ai_responses (
    id          BIGSERIAL PRIMARY KEY,
    event_id    BIGINT REFERENCES platform_events(id),
    platform    VARCHAR(20) NOT NULL,
    prompt      TEXT        NOT NULL,
    response    TEXT        NOT NULL,
    tokens_used INTEGER,
    created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Tabela de conteúdo sincronizado entre plataformas
CREATE TABLE IF NOT EXISTS content_sync_log (
    id          BIGSERIAL PRIMARY KEY,
    source_platform   VARCHAR(20) NOT NULL,
    target_platform   VARCHAR(20) NOT NULL,
    content_id        VARCHAR(100),
    status            VARCHAR(20) DEFAULT 'pending',
    created_at        TIMESTAMPTZ DEFAULT NOW()
);

-- Índices
CREATE INDEX IF NOT EXISTS idx_events_platform ON platform_events(platform);
CREATE INDEX IF NOT EXISTS idx_events_processed ON platform_events(processed);
CREATE INDEX IF NOT EXISTS idx_responses_event ON ai_responses(event_id);
```
