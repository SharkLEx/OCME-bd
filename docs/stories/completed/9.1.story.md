## 🎨 Design Spec (Sati)

### `design_tokens.py`

```python
# Brand — Manual da Marca WEbdEX (2024-07-18)
PINK_LIGHT  = 0xFB0491
RED_LIGHT   = 0xD90048
BLACK       = 0x000000
DARK        = 0x131313
WHITE       = 0xFFFFFF

# Semantic
SUCCESS     = 0x00FFB2
WARNING     = 0xFF8800
ERROR       = 0xFF4455
PRO_PURPLE  = 0xE040FB
CHART_BLUE  = 0x00D4FF

EMBED_COLORS = {
    "brand":      PINK_LIGHT,
    "status":     SUCCESS,
    "chart":      CHART_BLUE,
    "milestone":  PINK_LIGHT,
    "ciclo_win":  SUCCESS,
    "ciclo_loss": WARNING,
    "alert":      RED_LIGHT,
    "pro":        PRO_PURPLE,
    "gm":         PINK_LIGHT,
    "new_holder": RED_LIGHT,
    "default":    DARK,
}
```

### `embed_builder.py`

```python
import discord
from .design_tokens import EMBED_COLORS

FOOTER_TEXT = "WEbdEX Protocol · bdZinho"
BDZINHO_ICON = "https://i.ibb.co/MkcqbvLb/post-149-operador-da-tecnologia-01.jpg"

def build_embed(
    embed_type: str,
    title: str,
    description: str = "",
    fields: list[dict] = None,   # [{"name": str, "value": str, "inline": bool}]
    thumbnail: str = None,
    show_footer: bool = True,
) -> discord.Embed:
    color = EMBED_COLORS.get(embed_type, EMBED_COLORS["default"])
    embed = discord.Embed(title=title, description=description, color=color)
    for f in (fields or []):
        embed.add_field(name=f["name"], value=f["value"], inline=f.get("inline", False))
    if thumbnail:
        embed.set_thumbnail(url=thumbnail)
    if show_footer:
        embed.set_footer(text=FOOTER_TEXT, icon_url=BDZINHO_ICON)
    return embed
```
